//En este ejemplo vemos un caso de polimorfismo donde dos clases heredan de la clase animal un m√©todo pero modifican su comportamiento

class Animal {
  public void makeSound() {
    System.out.println("Grr...");
  }
}
class Cat extends Animal {
  public void makeSound() {
    System.out.println("Meow");
  }
}
class Dog extends Animal {
  public void makeSound() {
    System.out.println("Woof");
  }
}
